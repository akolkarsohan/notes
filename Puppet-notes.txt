# Checking Rescource declarations

  puppet resource <type> <title>
  puppet resource user root

# Checking status of resource package

  puppet resource package <packagename>
  puppet resource package ntp

# Parsing manifest file

  puppet parser validate filename.pp

# Apply the changes made to manifest file

  puppet apply filename.pp

# Print modulepath

  puppet config print modulepath

# Examine the directory structure of a module.

  tree <modulename>

# Gather list of facts of module

  facter <modulename>

# Install module from Forge (modules are placed in /etc/puppetlabs/code/environments/production/modules)

  puppet module install <modulename>

# Search module using a string

  puppet module search <string>
  puppet module search mysql

# Run puppet agent manually

  puppet agent -t
